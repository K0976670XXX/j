def check_password(password):
    length = len(password)
    if not(length>=8 and length<=10):#判斷是否為8~10位
        return 'Invalid password'#不符合直接結束
    if not(password.isalnum()):#判斷是否為數字和英文組成
        return 'Invalid password'
    count = 0#計算數字數量
    for a in password:
        if a.isdigit():#判斷是否為數字
            count+=1
    if length-count<1:#至少要包含有一字母，至少三個數字
        return 'Invalid password'
    if length<3:#至少要包含三個數字
        return 'Invalid password'
    return 'Valid password'

print(check_password(input('輸入密碼')))

def bubble_sort(R):
  print(f'氣泡排序法排序前')
  for i in range(10):
    for j in range(10-i-1):#最大的會不斷的移到最後
      if R[j]>R[j+1]:
        R[j],R[j+1]=R[j+1],R[j]
    print(f'第{i+1}回合:{R}')
  print(f'氣泡排序法排序後:\n{R}')
def select_sort(R):
  print(f'選擇排序前:\n{R}')
  for i in range(10):
    min_p = i #紀錄最小值的位置，初始化為欲排序位置
    for j in range(i+1,10):#從i的下一位開始往後找最小值
      if R[j]<R[min_p]:
        min_p = j#紀錄最小值位置
    R[i],R[min_p]=R[min_p],R[i]#位置互換
    print(f'第{i+1}回合:{R}')
  print(f'選擇排序後:\n{R}')

import numpy as np

def method_1(A):#矩陣擴張法較易用for loop實現
  p = 0#正方向
  n = 0#負方向
  s = 3#矩陣大小
  for i in range(s):
    cp,cn = 1,1#暫時存放
    for j in range(s):
      cp*=A[j][(i+j)%s]
      cn*=A[(i+s-1-j)%s][j]
    p+=cp
    n+=cn
  print(p-n)

def show_list(N):
  for i in range(3):
    for j in range(3):
      print(N[i][j],end=' ')
    print()
def method_1(N):
  total = []#存放總分
  avg = []#存放平均
  name = []#存放名字
  C = []#輸出
  for i in range(3):#計算總分和平均分
    for j in range(3):
      S = N[i][j][1]+N[i][j][2]+N[i][j][3]
      total.append(S)
      avg.append(round(S/3,2))
      name.append(N[i][j][0])
  for i in range(9):#做選擇排序
    max_p = i #紀錄最小值的位置，初始化為欲排序位置
    for j in range(i+1,9):#從i的下一位開始往後找最小值
      if total[j]>total[max_p]:
        max_p = j#紀錄最小值位置
    total[i],total[max_p]=total[max_p],total[i]#位置互換
    avg[i],avg[max_p]=avg[max_p],avg[i]
    name[i],name[max_p]=name[max_p],name[i]
  for i in range(3):
    D = []
    for j in range(3):
      D.append([f'#{3*i+j+1}',name[3*i+j],total[3*i+j],avg[3*i+j]])
    C.append(D)
  return C
def method_2(N):
  C = []#輸出
  D = []#為方便而建置的2維矩陣
  for i in range(3):#計算總分和平均分
    for j in range(3):
      S = N[i][j][1]+N[i][j][2]+N[i][j][3]
      D.append(['#0',N[i][j][0],S,round(S/3,2)])
  D = sorted(D,key=lambda T: T[2],reverse=True)#利用lambda運算子依照總分去排序，reverse=True:使之由大到小
  for i in range(3):#將結果轉換成三維矩陣
    E = []
    for j in range(3):
      D[3*i+j][0]=f'#{3*i+j+1}'#修改排名
      E.append(D[3*i+j])
    C.append(E)
  return C
